// دالة لتشغيل الأصوات function playSystemSound(text) { let audioFile = ""; if (text.includes("حلّ الظلام")) audioFile = "/sounds/night_start.mp3"; else if (text.includes("استيقظت المدينة")) audioFile = "/sounds/day_start.mp3"; else if (text.includes("الطبيب كان في المكان المناسب")) audioFile = "/sounds/saved.mp3"; else if (text.includes("انضم إلى اللعبة")) audioFile = "/sounds/join.mp3"; if (audioFile) { const audio = new Audio(audioFile); audio.play().catch(e => console.log("تحتاج للتفاعل مع الصفحة أولاً لتشغيل الصوت")); } }
